on_leave_court = {
	effect = {
		if = {
			limit = {
				has_trait = slave
				scope:old_employer = {
					is_ruler = yes
					has_relation_slave = prev
				}
			}
			scope:old_employer = {
				add_courtier = prev
			}
			debug_log = "Slave left court (and was brought back)"
			debug_log_date = yes
			debug_log_scopes = yes
		}
	}
}
