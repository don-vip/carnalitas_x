carnx_seize_slave_interaction = {
	category = interaction_category_prison
	desc = carnx_seize_slave_interaction_desc
	interface_priority = 100
	common_interaction = yes

	send_name = carnx_seize_slave_interaction

	ai_min_reply_days = 0
	ai_max_reply_days = 0

	is_shown = {
		NOT = { has_game_rule = carn_slavery_content_disabled }
		scope:actor = {
			NOT = { has_relation_slave = scope:recipient }
		}
		scope:recipient = {
			is_imprisoned_by = scope:actor
			has_trait = slave
			any_relation = {
				type = slave_owner
				exists = this
			}
		}
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			NOT = { has_strong_hook = scope:actor }
			custom_description = {
				text = "currently_being_tortured"
				NOT = { has_character_flag = is_being_tortured }
			}
		}
	}

	auto_accept = yes

	on_accept = {
		scope:recipient = {
			every_relation = {
				type = slave_owner
				hidden_effect = {
					remove_relation_slave = prev
					add_opinion = {
						target = prev
						modifier = carn_former_slave_or_slave_owner_opinion
					}
					reverse_add_opinion = {
						target = prev
						modifier = carn_former_slave_or_slave_owner_opinion
					}
				}
				add_opinion = {
					target = scope:actor
					modifier = carn_stole_my_slave_opinion
				}
			}
			set_relation_slave_owner = scope:actor
		}
		scope:actor = {
			hidden_effect = {
				send_interface_message = {
					type = event_generic_neutral
					title = carn_msg_enslave
					right_icon = scope:recipient
					custom_tooltip = carn_msg_scope_recipient_becomes_your_slave_desc
				}
			}
		}
		hidden_effect = {
			scope:actor = {
				if = {
					limit = {
						scope:recipient = { NOT = { is_courtier_of = scope:actor } }
					}
					add_courtier = scope:recipient
				}
			}
			scope:recipient = {
				if = {
					limit = { is_imprisoned_by = scope:actor }
					release_from_prison = yes
				}
				trigger_event = {
					on_action = carnx_on_slave_seized
				}
			}
		}
	}

	ai_targets = {
		ai_recipients = prisoners
	}

	ai_frequency = 24

	ai_potential = {
		is_at_war = no
	}

	ai_will_do = {
		base = 25

		# Opinion of prisoner's owner (-25 to 25 depending on opinion)
		opinion_modifier = {
			trigger = {
				scope:recipient = {
					any_relation = {
						type = slave_owner
						save_temporary_scope_as = owner
					}
				}
				exists = scope:owner
				NOT = { scope:actor = scope:owner }
			}
			opinion_target = scope:owner
			multiplier = -0.25
		}

		# Honor (-50 to 50 depending on honor)
		ai_value_modifier = {
			ai_honor = -0.5
		}

		# Greed (0 to Price depending on greed)
		modifier = {
			add = {
				value = ai_greed
				multiply = 0.01
				add = 1
				multiply = scope:recipient.carnx_slave_base_price_value
				divide = 2
			}
		}

		# Slavery shunned by faith (-75)
		modifier = {
			OR = {
				faith = {
					has_doctrine = carn_doctrine_same_slavery_shunned
					faith_hostility_level = {
						target = scope:recipient.faith
						value < faith_hostile_level
					}
				}
				faith = {
					has_doctrine = carn_doctrine_other_slavery_shunned
					faith_hostility_level = {
						target = scope:recipient.faith
						value >= faith_hostile_level
					}
				}
			}
			add = -75
		}

		# Family and relations (50 to 200 depending on how close the relation)
		# Want to seize close relations in order to free them
		carnx_family_and_relations_modifier = {
			RULER = scope:actor
			SLAVE = scope:recipient
			SCALE = 50
		}

		# Lacking councillors, court physician, or knights (2 for each relevant skill above 12)
		carnx_lacking_councillors_modifier = {
			RULER = scope:actor
			SLAVE = scope:recipient
			SCALE = 2
		}
		carnx_lacking_court_physician_modifier = {
			RULER = scope:actor
			SLAVE = scope:recipient
			SCALE = 2
		}
		carnx_lacking_knights_modifier = {
			RULER = scope:actor
			SLAVE = scope:recipient
			SCALE = 2
		}

		# Claims (25 if useful claims on neighbouring realms)
		carnx_claims_modifier = {
			RULER = scope:actor
			SLAVE = scope:recipient
			SCALE = 25
		}

		# Attraction (0 to 25 depending on attraction, doubled for lustful, halved for chaste)
		carnx_attraction_modifier = {
			RULER = scope:actor
			SLAVE = scope:recipient
			SCALE = 25
		}

		# Female (25 if a visibly fertile female)
		carnx_female_modifier = {
			SLAVE = scope:recipient
			SCALE = 25
		}

		# Execute reason, rival, cheating spouse (75) or nemesis (150)
		carnx_execute_reason_modifier = {
			RULER = scope:actor
			SLAVE = scope:recipient
			SCALE = 75
		}

		# Don't seize if execute reason and recipient imprisoned for less than 2 years
		# Prefer executing instead of seizing if AI is up to it
		modifier = {
			carnx_execute_reason_trigger = yes
			scope:recipient = {
				time_in_prison = { years < 2 }
			}
			factor = 0
		}
	}
}
