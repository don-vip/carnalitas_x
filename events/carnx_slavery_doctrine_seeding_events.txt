namespace = carnx_slavery_doctrine_seeding

# Event chain for seeding religions with slavery tenet doctrines

carnx_slavery_doctrine_seeding.0001 = {
	type = empty
	hidden = yes

	immediate = {
		if = {
			limit = {
				has_game_rule = carn_slavery_default_religious_doctrine_full_historical
			}
			every_religion_global = {
				every_faith = {
					if = {
						limit = {
							NOT = { has_doctrine = carnx_special_doctrine_slave_employment }
						}
						if = {
							limit = {
								religion = religion:islam_religion
							}
							add_doctrine = carnx_special_doctrine_slave_employment
						}
					}
				}
			}
		}
	}
}
